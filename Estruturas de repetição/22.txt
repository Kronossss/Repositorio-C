#include <stdio.h>

int main()
{
    float A[40][2],o,r;
    int i,g,l;
    o=0;
    l=0;
    for(i=0; i<5; i++){
        for(g=0; g<2; g++){
            printf("Digite a %iº nota do %i aluno : ",g+1,i+1);
            scanf("%f",&A[i][g]);
            o+=A[i][g];
        }
    }
    r=o/10;
    for(i=0; i<5; i++){
       if((A[i][0]>r)||(A[i][1]>r)){
        l+=1;   
       } 
    }
    printf("Número de alunos que tiraram a nota acima da média : %i",l);
    

    return 0;
}
