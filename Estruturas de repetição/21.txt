
	#include <stdio.h>

int main()
{
    float A[40][2],o,r;
    int i,g;
    o=0;
    for(i=0; i<40; i++){
        for(g=0; g<2; g++){
            printf("Digite a %iº nota do %i aluno : ",g+1,i+1);
            scanf("%f",&A[i][g]);
            o+=A[i][g];
        }
    }
    r=o/80;
    printf("Essa é a média da turma : %.2f",r);

    return 0;
}
