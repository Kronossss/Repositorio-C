#include <stdio.h>

int main()
{
    int A[20],i,g,o;
    o=19;
    for (i=0; i<20; i++){
    printf("Digite o %iº número do vetor : ",i+1);
    scanf("%i",&A[i]);
    }
    for (i=0; i<20; i++){
    printf("%iº número do vetor  : %i \n",i+1,A[i]);
    }

    }

    }
    return (0);
}
