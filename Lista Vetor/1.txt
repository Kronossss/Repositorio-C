#include <stdio.h>

int main()
{
    int A[20],i,g;
    g=0;
    for (i=0; i<20; i++){
       printf("Digite o %iº número do vetor : ",i+1);
       scanf("%i",&A[i]);
       if(A[i]%2==0){
           g+=1;
       }
    }
    for(i=0; i<20; i++){
        printf("%iº número do vetor : %i \n",i+1,A[i]);
    }
    printf("Essa é a quantidade de números pares no vetor : %i",g);

    return 0;
}
