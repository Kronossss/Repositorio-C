struct RACIONAL {
int numerator;
int denominator;
};

#include <stdio.h>

int main()
{
    struct RACIONAL Dif;
    int soma,sub;
    float divi;
    printf("Digite o numerador : ");
    scanf("%i",&Dif.numerator);
    printf("Digite o denominator : ");
    scanf("%i",&Dif.denominator);
    soma=Dif.numerator+Dif.denominator;
    sub=Dif.numerator-Dif.denominator;
    divi=(float)Dif.numerator/(float)Dif.denominator;
    printf("O resultado da soma é : %i\n",soma);
    printf("O resultado da subtração é : %i\n",sub);
    printf("O resultado da divisão é : %.2f \n",divi);
    return 0;
}
