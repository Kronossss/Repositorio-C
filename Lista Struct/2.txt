struct Escola{
int Mat[10];
float Nota[10][20];
float Med[10];
};

#include <stdio.h>

int main()
{
    struct Escola Alunos;
    int i,g;
    for(i=0; i<10; i++){
        Alunos.Med[i]=0;
    }
    for(i=0; i<10; i++){
        printf("Digite a matrícula do %iº aluno : ",i+1);
        scanf("%i",&Alunos.Mat[i]);
        for(g=0; g<2; g++){
        printf("Digite a %iº nota do %iº aluno : ",g+1,i+1);
        scanf("%f",&Alunos.Nota[i][g]);
        Alunos.Med[i]+=Alunos.Nota[i][g];
        }
    }
    for(i=0; i<10; i++){
        printf("%iº aluno : Matricula :%i Média da nota :%.2f\n",i+1,Alunos.Mat[i],Alunos.Med[i]/2);
    }
    return 0;
}
